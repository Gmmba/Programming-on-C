#include  <stdio.h>
int main() 
{
float a;
char b;
printf("Введите температуру и единицу измерения('C' Цельсий или 'F' Фаренгейт): ");
scanf("%e", &a);
scanf("%c", &b);

} else if (b == 'F'|| b == 'f') {
	printf("%f", (a-32)/1.8);

}
}
